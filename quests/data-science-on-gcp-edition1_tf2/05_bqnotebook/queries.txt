SELECT
  ORIGIN,
  AVG(DEP_DELAY) AS dep_delay,
  AVG(ARR_DELAY) AS arr_delay
FROM
  `cloud-training-demos.flights.fedtzcorr`
GROUP BY
  ORIGIN
ORDER BY
  arr_delay DESC
